- Data-model `MultiPhaseAI` (3-phase progressive trigger overlap) :
  ```ts
  type AbilityTrigger = {
    hpRange: { min: number; max: number }; // % HP
    abilityId: string;
  };
  // Forest Runner :
  // - Piercing Beak : { min: 0.5, max: 1.0 }
  // - Menacing : { min: 0.25, max: 0.5 }
  // - Wooing : { min: 0, max: 0.5 }

  // = 25-50% : both eligible (equal chance)
  // = >50% : Piercing Beak only
  ```

  ```ts
  type Ability = {
    statusProc?: {
      statusId: 'bewitchment' | 'dispirit' | ...;
      baseChance: number; // 0-1
      avReducedBy?: 'physical' | 'magic'; // A-AV vs M-AV reduces
    };
  };
  // Wooing : { statusId: 'bewitchment', baseChance: 1.0, avReducedBy: 'magic' }
  ```
